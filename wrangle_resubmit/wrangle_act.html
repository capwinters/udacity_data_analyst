Source 3: details¶

Note: Since it takes long time, the following cell is ran only once and given as markdown below. I gathered the required info and wrote it to a txt file. While working on the data I am reading "tweet_json.txt" into a dataframe called `details`.
consumer_key = None
consumer_secret = None
access_token = None
access_secret = None

# authenticate twitter api
auth = tweepy.OAuthHandler(consumer_key, consumer_secret)
auth.set_access_token(access_token, access_secret)
api = tweepy.API(auth, wait_on_rate_limit=True, wait_on_rate_limit_notify = True)

# we will append info from found tweets into this list 
tweet_details = []

# count how many we miss
skipped_tweet_cnt = 0

# loop on the tweet ids from archive file 
for tweet_id in archive.tweet_id:

    # shall not crash if get_status gives an exception 
    # we store as much as we can get    
    try:
        tweet = api.get_status(tweet_id)
        tweet_details.append( dict([("id",tweet._json["id"])
                                ,("favorite_count", tweet._json["favorite_count"])
                                ,("retweet_count", tweet._json["retweet_count"])]))

    except tweepy.TweepError as e:
        print (str(type(e)) + "-" + str(e.args))          
        skipped_tweet_cnt += 1
        pass

print(skipped_tweet_cnt)
retweets_and_favs = pd.DataFrame(tweet_details)

# csv is also a text file. extention does not matter as long as we know it is comma spsrated while reading. 
# i could use any other delimiteri but preferred with csv for simplicity.
retweets_and_favs.to_csv("tweet_json.txt")

def set_stage(doggo, floofer, pupper, puppo): 
    
    if pd.isnull(doggo):     
        
        if pd.isnull(floofer):
            
            if pd.isnull(pupper):
                
                if pd.isnull(puppo):
                    
                    return None
                else:
                    return puppo
            else:
                return pupper
        else:
            return floofer
    else:
        return doggo
